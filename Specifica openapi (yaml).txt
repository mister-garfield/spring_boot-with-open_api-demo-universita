components:
  schemas:
    # Modelli ed entita
    Persona:
      type: object
      properties:
        id:
          type: integer
          format: int32
          readOnly: true
          description: Sarà univoco in quanto chiave primaria (pk) nel database.
          minimum: 0
        nome:
          type: string
        cognome:
          type: string
        codiceFiscale:
          type: string
          description: |
            Deve essere univoco. 
            Codice che identifica una persona a livello nazionale in base a nome, cognome e data di nascita. 
            E costruito secondo le seguenti regole: 16 caratteri, solo lettere A-Z e numeri 0-9 ordinati secondo le regole dell'agenzia delle entrate.
          minLength: 16
          maxLength: 16
          pattern: ^[A-Z]{6}[0-9]{2}[A-Z][0-9]{2}[A-Z][0-9]{3}[A-Z]
        sesso:
          type: string
          enum:
          - N
          - F
          - M
          description: |
            Deve essere 1 sola lettera. 
            Puo avere solo 3 valori: N sta per neutro, F per femmina, M per maschio.
        tipo: 
          type: string
      discriminator:
        propertyName: tipo
        mapping:
          studente: '#/components/schemas/Studente'
          docente: '#/components/schemas/Docente'
      required:
        - nome
        - cognome
        - codiceFiscale
        - sesso
        - tipo
      examples:
      - id: 0
        nome: Mario
        cognome: Rossi
        codiceFiscale: RSSMRA80A01H501U
        sesso: M
        tipo: studente / docente
    Studente:

          maxLength: 7

          type: string

        tipo:
          type: string
          default: studente
          pattern: ^studente$
          readOnly: true
      required:
        - matricola
        - corsoDiLaurea
        - tipo
      examples:
